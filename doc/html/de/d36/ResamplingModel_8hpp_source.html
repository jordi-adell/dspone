85 DSPONE_REASMPLING_MODEL_TMP
86 void DSPONE_RESAMPLING_MODEL_DCL::resample(const T_ParticleSet &currentSet,
87  T_ParticleSet &resampledSet,
88  const T_WeightSet &weights)
89 {
90 
91  int size = currentSet.size();
92  if (size != resampledSet.size() || weights.size() != size)
93  {
94  ERROR_STREAM("You provided sets with different sizes, so the particle set has not been resampled");
95  return;
96  }
97 
98  T_Weight total = weights.sum();
99  int resampledSize = size*0.99;
100 
101  // This plus one, is to avoid step to be equal to cumSum, bacause if so
102  // then the last particle in the set is always included despite the value
103  // of its corresponding weight.
104  T_Weight step = total / (resampledSize + 1);
105  if (step <= 0)
106  {
107  ERROR_STREAM("Step in particle filter resampling is negative or zero, particle filter is not realiable now");
108  return;
109  }
110 
111  T_Weight cumSum = 0;
112  T_Weight steps = step;
113  int istep = 0;
114  for (int iparticle = 0; iparticle < size; ++iparticle)
115  {
116  cumSum += weights.at(iparticle);
117  while (steps < cumSum)
118  {
119  resampledSet.set(currentSet.at(iparticle), istep);
120  steps += step;
121  ++istep;
122  }
123  }
124 
125  for (; istep < resampledSize; ++istep)
126  {
127  resampledSet.set(currentSet.at(currentSet.size() -1), istep);
128  }
129 
130  for (; istep < size; ++istep)
131  {
132  resampledSet.set(resampledSet.getRandomParticle() ,istep);
133  }
134 }
