119  DSPONE_PREDICTION_MODEL_TMP
120  void DSPONE_PREDICTION_MODEL_DCL::updateModel(T_Particle solution)
121  {
122  // Low pass iir filter to the velocity
123  // v[n+1] = factor*v_estimated + (1-factor)*v[n-1]
124  T_Particle prevVel = _velocity;
125  _velocity *= (1 - _velocityUpdateFactor);
126  _velocity += _velocityUpdateFactor * (solution - _prevParticle);
127  TRACE_STREAM("Vel: " << _velocity << " = (" << solution << " - " << _prevParticle << ")" << " update factor: " << _velocityUpdateFactor);
128  _prevParticle = solution;
129  if (_velocity > prevVel)
130  _velocityUpdateFactor = sqrt(_velocityUpdateFactor);
131  else
132  _velocityUpdateFactor *= _velocityUpdateFactor;
133  _velocityUpdateFactor = std::max(_velocityUpdateFactor, MIN_VELOCITY_UPDATE_FACTOR);
134  _velocityUpdateFactor = std::min(_velocityUpdateFactor, MAX_VELOCITY_UPDATE_FACTOR);
135  }
