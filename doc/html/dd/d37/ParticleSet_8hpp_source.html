208 void DSPONE_BASIC_PARTICLE_SET_DCL::normalise()
209 {

211  // possitive.

214  math::subC(min, _set.get(), IParticleSet<T_Particle>::_size);

226 DSPONE_BASIC_PARTICLE_SET_TMP
227 void DSPONE_BASIC_PARTICLE_SET_DCL::addNoise(T_Particle mean, T_Particle std)
228 {
229  math::rand(_randState, _randomValues.get(), IParticleSet<T_Particle>::_size);
230  math::multC(std, _randomValues.get(), IParticleSet<T_Particle>::_size);
231  math::addC(mean, _randomValues.get(), IParticleSet<T_Particle>::_size);
232  math::copy(_randomValues.get(), _randomParticles.get(), IParticleSet<T_Particle>::_size);
233  math::add(_randomParticles.get(), _set.get(), IParticleSet<T_Particle>::_size);
234 }
